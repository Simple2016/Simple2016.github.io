@Override
public void doFilter(ServletRequest req, ServletResponse res, FilterChain chain) throws IOException, ServletException {

    HttpServletRequest request = (HttpServletRequest) req;
    HttpServletResponse response = (HttpServletResponse) res;

    response.setHeader("Access-Control-Allow-Origin","http://123.123.123.123:123");
    response.setHeader("X-Frame-Options","allow-from http://123.123.123.123:123");
    response.setHeader("Content-Security-Policy","frame-ancestors http://123.123.123.123:123");

    response.setHeader("Access-Control-Allow-Credentials","true");
    response.setHeader("Access-Control-Allow-Headers","appkey, Content-Type, User-Agent");
    response.setHeader("Access-Control-Allow-Methods","POST,OPTIONS,GET");
    //System.out.println(request.getMethod());
    if ("OPTIONS".equals(request.getMethod())){
        response.setStatus(HttpStatus.SC_OK);
    }

    chain.doFilter(req, res);
}

没时间写具体找问题的过程了，直接上结果，主要是看浏览器请求发现是 option类型的请求，然后用postman 直接请求options到服务端，直接403返回了tomcat 的错误页面，后台根本没有收到options类型的请求。所以找到问题的终极病根是为什么options 的请求被拦截了

※ 解决

说明一下，首先这个项目比较旧了，而且东西杂乱，具体的再深入的原因暂不追究，以下是这次遇到的问题的解决方法，新的springboot项目也许根本不存在这些问题的

以下是这次遇到的两个问题的解决方法，第二个问题解决比较麻烦，花的时间也最多

  1. 参数问题可能会导致跨域

    看到他们传了json 。所以我后台也是要接收的，不然可能存在跨域问题

  2. 参数中自定义了headers，导致发送了预请求。

    以下下情况会发送预请求：

    1:请求的方法不是GET/HEAD/POST

    2:POST请求的Content-Type并非application/x-www-form-urlencoded, multipart/form-data, 或text/plain

    3:请求设置了自定义的header字段

    来自 https://www.cnblogs.com/wanghuijie/p/preflighted_request.html

    解决方法

    1. 打开web.xml

    2. 首先低版本的spring 是默认不接收options 请求的，

      需要开启

      <servlet-class>org.springframework.web.servlet.DispatcherServlet</servlet-class>
<!--在上边的位置加上下边的代码-->
<init-param>
     <param-name>dispatchOptionsRequest</param-name>
     <param-value>true</param-value>
</init-param>

    3. 安全约束(如果开启的话)中放开 OPTIONS

      <security-constraint>
    <web-resource-collection>
        <web-resource-name>fortune</web-resource-name>
        <url-pattern>/*</url-pattern>
        <http-method>PUT</http-method>
        <http-method>DELETE</http-method>
        <http-method>HEAD</http-method>
        <!--<http-method>OPTIONS</http-method>-->
        <http-method>TRACE</http-method>
    </web-resource-collection>
    <auth-constraint />
</security-constraint>

    4. 服务端添加判断（参考最上边的代码）

       if ("OPTIONS".equals(request.getMethod())){
        response.setStatus(HttpStatus.SC_OK);
 }

iframe 到底有没有跨域问题

分情况： 只是嵌入看x-frame-options是否允许嵌入，如果允许嵌入的话，嵌入没有问题
然后，如果要在父页面访问iframe里的东西，需要考虑跨域问题。

跨域了有很多的解决方案。可以参考 https://www.jianshu.com/p/b1de3e05bfb6

跨域问题的解决方法的本质，就是你跨域访问的资源允许你跨域访问了，你才能访问成功，否则根据同源策略，是不可能让一个陌生的网页直接获取本网页的数据的，这是同源策略的规范，浏览器实现了这个规范，为了网站的安全。

注意1 ： ajax 跨域请求的时候，contentType 也会影响跨域。所以如果出现问题可以考虑contentType 是否与资源匹配，或者直接去掉。ps: 这个问题一度导致我一度怀疑人生，在后端配置好跨域后却依旧提示跨域访问。花费了大约4个小时才发现是这个问题。

注意2 : ajax 的其他参数也可能引起跨域问题，不只是contentType ！！！

注意3 ： data 参数不对也会有问题！！！

注意4 ： 浏览器直接打开这个文件也是可以的，这时候ajax访问任何页面都算是跨域，不过跨域提示的文字显示本地址为null

Access to XMLHttpRequest at ‘https://www.baidu.com/' (redirected from ‘http://www.baidu.com/') from origin ‘null’ has been blocked by CORS policy: No ‘Access-Control-Allow-Origin’ header is present on the requested resource.

所以如果测试某个ip的话，建议把这个页面放到容器中，tomcat 或者iis 都可

1，创建一个基本的返回结果类

public class HttpResult {
    //状态码
    final public int code;
    //内容
    final public String content;


    public HttpResult(int code, String content) {
        this.code = code;
        this.content = content;
    }

2 get 和post 的写法。


import org.apache.http.Consts;
import org.apache.http.NameValuePair;
import org.apache.http.client.config.RequestConfig;
import org.apache.http.client.entity.UrlEncodedFormEntity;
import org.apache.http.client.methods.CloseableHttpResponse;
import org.apache.http.client.methods.HttpGet;
import org.apache.http.client.methods.HttpPost;
import org.apache.http.impl.client.CloseableHttpClient;
import org.apache.http.impl.client.HttpClients;
import org.apache.http.message.BasicNameValuePair;
import org.apache.http.util.EntityUtils;

import java.io.IOException;
import java.util.ArrayList;
import java.util.List;
import java.util.Map;

public class RequestClient {


    /**
     * 带参数的post请求
     *
     * @param url
     * @param map
     * @return
     * @throws Exception
     */
    public static HttpResult doPost(String url, Map<String, Object> map) throws Exception {
        CloseableHttpClient httpClient = HttpClients.createDefault();
        // 声明httpPost请求
        HttpPost httpPost = new HttpPost(url);
        RequestConfig requestConfig = RequestConfig.custom()
                .setConnectTimeout(20000).setConnectionRequestTimeout(20000)
                .setSocketTimeout(20000).build();
        httpPost.setConfig(requestConfig);
        // 判断map不为空
        if (map != null) {
            // 声明存放参数的List集合
            List<NameValuePair> params = new ArrayList<NameValuePair>();

            // 遍历map，设置参数到list中
            for (Map.Entry<String, Object> entry : map.entrySet()) {
                params.add(new BasicNameValuePair(entry.getKey(), entry.getValue().toString()));
            }

            // 创建form表单对象
            UrlEncodedFormEntity formEntity = new UrlEncodedFormEntity(params, "utf-8");

            // 把表单对象设置到httpPost中
            httpPost.setEntity(formEntity);
        }

        // 使用HttpClient发起请求，返回response
        CloseableHttpResponse response = httpClient.execute(httpPost);

        // 解析response封装返回对象httpResult
        HttpResult httpResult = getHttpResult(response);
        // 返回结果
        return httpResult;
    }

    /**
     * 带参数的get请求
     *
     * @param url
     * @param map
     * @return
     * @throws Exception
     */
    public static HttpResult doGet(String url, Map<String, Object> map) throws Exception {
        CloseableHttpClient httpClient = HttpClients.createDefault();
        String paramsStr = "";
        // 判断map不为空
        if (map != null) {
            // 声明存放参数的List集合
            List<BasicNameValuePair> params = new ArrayList<BasicNameValuePair>();

            // 遍历map，设置参数到list中
            for (Map.Entry<String, Object> entry : map.entrySet()) {
                params.add(new BasicNameValuePair(entry.getKey(), entry.getValue().toString()));
            }
            paramsStr = EntityUtils.toString(new UrlEncodedFormEntity(params, Consts.UTF_8));
        }
        // 声明httpGet请求
        HttpGet httpGet = new HttpGet(url + "?" + paramsStr);
        RequestConfig requestConfig = RequestConfig.custom()
                .setConnectTimeout(20000).setConnectionRequestTimeout(20000)
                .setSocketTimeout(20000).build();
        httpGet.setConfig(requestConfig);

        // 使用HttpClient发起请求，返回response
        CloseableHttpResponse response = httpClient.execute(httpGet);

        // 解析response封装返回对象httpResult
        HttpResult httpResult = getHttpResult(response);
        // 返回结果
        return httpResult;
    }

    private static HttpResult getHttpResult(CloseableHttpResponse response) throws IOException {
        HttpResult httpResult = null;
        if (response.getEntity() != null) {
            httpResult = new HttpResult(response.getStatusLine().getStatusCode(),
                    EntityUtils.toString(response.getEntity(), "UTF-8"));
        } else {
            httpResult = new HttpResult(response.getStatusLine().getStatusCode(), "");
        }
        return httpResult;
    }
}

上班的路上，冷空气让鼻子变得酸胀，同时也变得灵敏了起来，路过草坪，泥土和青草的味道闻起来格外的浓重。不禁让人回忆连篇，所有在这个季节发生的事情仿佛就在昨天一样。想起来三年前的这个时候，是我刚到上家公司工作的时候，时间真是一眨眼的功夫，让人不能相信，又不能不相信。

曾经我如同我们的青春一样怀揣着梦想，相信所有美好的事情终将发生在自己的将来。然而一天又一天过去了，梦想逐渐被尘土掩埋。不是不知道自己的过得是不是以前期望的生活，而是自己根本就不再那么相信也不记得之前的期望了。

或许只有相貌才能说明我还是之前的我。听说人全身细胞每七年就全部更换一次。那些年的梦想，和自己真实的感受，永远的停留在了那里。偶尔会被一些外部环境激发出来，让人心生感慨。
