//获取当前时间，格式YYYY-MM-DD
function getNowFormatDate() {
    var myDate = new Date();
    var nowY = myDate.getFullYear();
    var nowM = myDate.getMonth() + 1;
    var nowD = myDate.getDate();
    var enddate = nowY + "-" + (nowM < 10 ? "0" + nowM : nowM) + "-" + (nowD < 10 ? "0" + nowD : nowD);//当前日期
    return enddate;
}

判断数组中数据是否重复

并忽略大小写


function isRepeatIgnoreCase(arr){  
       var hash = {};  
       for(var i in arr) {  
           if(hash[arr[i].toUpperCase()])  
                return true;  
           hash[arr[i].toUpperCase()] = true;  
       }  
       return false; 
} 
// 调用示例
isRepeat(['11010119990501625X','11010119990501625x'])

判断一个字符串是不是json

function isJSON(str) {
    if (typeof str == 'string') {
        try {
            var obj=JSON.parse(str);
            if(typeof obj == 'object' && obj ){
                return true;
            }else{
                return false;
            }

        } catch(e) {
            console.log('error：'+str+'!!!'+e);
            return false;
        }
    }
    console.log('It is not a string!')
}

Array.prototype.removeAll = function(val) { 
    for(var i=0;i<this.length;i++){
        var index = this.indexOf(val); 
        if (index > -1) { 
            this.splice(index, 1); 
        } else{
            break;
        }
    }
}

jquery 里的promise 写法

需要注意的点是

只针对单独的方法， a方法里边的调用b方法,b 方法依然是异步执行的

1 2 3 步是必须写的 缺一不可

function runAsync(){
    var defer = $.Deferred();//1步
    //做一些异步操作
    setTimeout(function(){
        console.log('执行完成');
        defer.resolve('异步请求成功之后返回的数据');//2步
    }, 5000);
    return defer;//3步
}
runAsync().then(function(data){
    //上边返回的数据
    alert(data);
});
