判断数组中数据是否重复

并忽略大小写


function isRepeatIgnoreCase(arr){  
       var hash = {};  
       for(var i in arr) {  
           if(hash[arr[i].toUpperCase()])  
                return true;  
           hash[arr[i].toUpperCase()] = true;  
       }  
       return false; 
} 
// 调用示例
isRepeat(['11010119990501625X','11010119990501625x'])
